2. 路径操作
    $ pfName=/home/sgs/test.txt

    # 获取全路径文件名
    $ echo $pfName
    /home/sgs/test.txt

    # 获取文件名
    $ echo ${pfName##*/}
    test.txt

    # 获取该文件的位置
    $ echo ${pfName%/*}
    /home/sgs

    # 获取文件的后缀名
    $ echo ${pfName##*.}
    txt

    # 仅仅获取文件的前缀
    $ echo ${pfName##*/} | sed -r 's/^(.*)\..*$/\1/g'
    test

3. find结果如何传递给shell数组

    array=($(find . -name "*.txt"))
    for i in "${array[@]}"; do echo $i; done

4. 传接命令的状态判断
    ls -al file.ext
    rc=$?; if [[ $rc != 0 ]]; then exit $rc; fi

    ${PIPESTATUS[0]} pipe line 第0个命令的exit code
    ${PIPESTATUS[*]} pipe line all exit code
